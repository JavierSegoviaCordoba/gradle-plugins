152 private fun String.addChanges(header: String, changes: List<String>): Changelog =
153     buildList<String> {
154             val firstVersionIndex =
155                 lines().indexOfFirst {
156                     it.startsWith("## [") && it.contains("[Unreleased]", true).not()
157                 }
158             var shouldAddUpdate = true
159             lines().onEach { line ->
160                 if (line.startsWith(header) && shouldAddUpdate) {
161                     shouldAddUpdate = false
162                     add(line)
163                     for (change in changes) {
164                         if (lines().subList(0, firstVersionIndex).none { it.contains(change) }) {
165                             add("- $change")
166                         }
167                     }
168                 } else {
169                     add(line)
170                 }
171             }
172             runCatching {
173                 forEachIndexed { index: Int, line ->
174                     val updateRegex = """(- `)(.*)( )(->)( )(.*)(`)"""
175                     if (Regex(updateRegex).matches(line)) {
176                         for (j in index + 1 until firstVersionIndex) {
177                             val lineToRemove = this[j]
