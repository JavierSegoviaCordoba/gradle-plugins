internal enum class HubdleKotlinMultiplatformTarget(val target: String) {
    Apple(target = "apple"),
    Common(target = "common"),
    Android(target = "android"),
    Jvm(target = "jvm"),
    JvmAndAndroid(target = "jvmAndAndroid"),
    IOS(target = "ios"),
